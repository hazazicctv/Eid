      type="text"

    <div class="buttons">
      <button onclick="makeCard()">إنشاء البطاقة | Generate Card</button>
      <button onclick="downloadCard()">تحميل البطاقة | Download Card</button>
    </div>

    <div id="status">جاري تحميل البطاقة... | Loading card...</div>

    <canvas id="c"></canvas>
  </div>

  <script>
    const imageFileName = "EidHazazi.png"; // تأكد من الاسم بالضبط
    const img = new Image();
    const canvas = document.getElementById("c");
    const ctx = canvas.getContext("2d");
    const statusEl = document.getElementById("status");

    let imageReady = false;

    img.onload = function () {
      imageReady = true;
      canvas.width = img.naturalWidth;
      canvas.height = img.naturalHeight;
      drawBaseImage();
      statusEl.textContent = "البطاقة جاهزة | Card is ready";
    };

    img.onerror = function () {
      statusEl.textContent =
        "تعذر تحميل الصورة. تأكد أن اسمها EidHazazi.png ومرفوعة بنفس مكان ملف index.html";
      alert("الصورة لم تتحمل. تأكد من اسم الصورة ومكانها.");
    };

    img.src = imageFileName;

    function drawBaseImage() {
      if (!imageReady) return;
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
    }

    function makeCard() {
      const name = document.getElementById("name").value.trim();

      if (!imageReady) {
        alert("الصورة لم تجهز بعد");
        return;
      }

      if (!name) {
        alert("اكتب اسمك أولاً | Please enter your name first");
        return;
      }

      drawBaseImage();

      let fontSize = 54;
      if (name.length > 14) fontSize = 44;
      if (name.length > 22) fontSize = 36;

      ctx.fillStyle = "black";
      ctx.textAlign = "center";
      ctx.textBaseline = "middle";
      ctx.font = 700 ${fontSize}px Arial;

      const x = canvas.width / 2;
      const y = canvas.height * 0.77; // نزلت الاسم أكثر

      ctx.fillText(name, x, y);
    }

    function downloadCard() {
      if (!imageReady) {
        alert("الصورة لم تجهز بعد");
        return;
      }

      const link = document.createElement("a");
      link.download = "eid-card.png";
      link.href = canvas.toDataURL("image/png");
      link.click();
    }
  </script>
