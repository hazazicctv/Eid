    <input id="name" type="text" placeholder="اكتب اسمك هنا | Enter your name here">

    <div class="buttons">
      <button onclick="makeCard()">إنشاء البطاقة | Generate Card</button>
      <button onclick="downloadCard()">تحميل البطاقة | Download Card</button>
    </div>

    <canvas id="c"></canvas>

    <div class="note">
      الاسم يدعم العربية والإنجليزية | Arabic and English names are supported

  <script>
    const img = new Image();
    img.src = "EidHazazi.png";

    const canvas = document.getElementById("c");
    const ctx = canvas.getContext("2d");

    let imageReady = false;

    img.onload = () => {
      canvas.width = img.width;
      canvas.height = img.height;
      ctx.drawImage(img, 0, 0);
      imageReady = true;
    };

    function drawBaseImage() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      ctx.drawImage(img, 0, 0);
    }

    function makeCard() {
      const name = document.getElementById("name").value.trim();

      if (!imageReady) {
        alert("الصورة لم تجهز بعد، جرّب مرة ثانية بعد لحظة.\nImage is still loading. Please try again.");
        return;
      }

      if (!name) {
        alert("اكتب اسمك أولاً.\nPlease enter your name first.");
        return;
      }

      drawBaseImage();

      ctx.fillStyle = "black";
      ctx.textAlign = "center";
      ctx.textBaseline = "middle";

      let fontSize = 52;
      if (name.length > 14) fontSize = 42;
      if (name.length > 22) fontSize = 34;

      ctx.font = 700 ${fontSize}px "Segoe UI", Tahoma, Arial, sans-serif;

      const x = canvas.width / 2;

      // نزلت الاسم أكثر من قبل
      const y = canvas.height * 0.745;

      ctx.fillText(name, x, y);
    }

    function downloadCard() {
      if (!imageReady) {
        alert("الصورة لم تجهز بعد.\nImage is still loading.");
        return;
      }

      const link = document.createElement("a");
      link.download = "eid-card.png";
      link.href = canvas.toDataURL("image/png");
      link.click();
    }
  </script>
