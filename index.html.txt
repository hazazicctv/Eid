  <button onclick="generateImage()">إنشاء البطاقة</button>
  <button onclick="downloadImage()">تحميل</button>

  <br>
  <canvas id="canvas"></canvas>

  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const image = new Image();
    image.crossOrigin = "anonymous"; // مهم
    image.src = "EidHazazi.png";

    let imageLoaded = false;

    image.onload = function () {
      canvas.width = image.width;
      canvas.height = image.height;
      ctx.drawImage(image, 0, 0);
      imageLoaded = true;
    };

    function generateImage() {
      const name = document.getElementById("nameInput").value.trim();

      if (!imageLoaded) {
        alert("الصورة لسه ما تحميلت، انتظر لحظة وجرب مرة ثانية");
        return;
      }

      if (!name) {
        alert("اكتب اسمك أولاً");
        return;
      }

      // رسم الصورة من جديد
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      ctx.drawImage(image, 0, 0);

      // إعداد النص
      ctx.fillStyle = "black";
      ctx.textAlign = "center";

      // حجم الخط حسب طول الاسم
      let fontSize = 60;
      if (name.length > 15) fontSize = 45;
      if (name.length > 25) fontSize = 35;

      ctx.font = "bold " + fontSize + "px Arial";

      // مكان الاسم (مضبوط لتحت Eid Mubarak)
      const x = canvas.width / 2;
      const y = canvas.height * 0.70;

      ctx.fillText(name, x, y);
    }

    function downloadImage() {
      if (!imageLoaded) {
        alert("اضغط إنشاء البطاقة أولاً");
        return;
      }

      const link = document.createElement("a");
      link.download = "eid-card.png";
      link.href = canvas.toDataURL("image/png");
      link.click();
    }
  </script>
